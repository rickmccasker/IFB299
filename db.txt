CREATE DATABASE smartcity;

CREATE TABLE `smartcity`.`services` (
  
	`id` INT NOT NULL AUTO_INCREMENT,
  
	`name` VARCHAR(45) NULL,
  
	`desc` VARCHAR(120) NULL,
  
	`city` VARCHAR(5) NULL,
  
	PRIMARY KEY (`id`));
